column_names = {'Date': 'date', 'Start (ET)': 'start',
                'Unamed: 2': 'box', 'Visitor/Neutral': 'away_team', 
                'PTS': 'away_points', 'Home/Neutral': 'home_team',
                'PTS.1': 'home_points', 'Unamed: 7': 'n_ot'}

games = (games.rename(columns=column_names) 
    .dropna(thresh=4) #dropna removes missing values
    [['date', 'away_team', 'away_points', 'home_team', 'home_points']]
    .assign(date=lambda x: pd.to_datetime(x['date'], format='%a, %b %d, %Y')) # 'assign'  assigns new columns to a DataFrame, which returns a new object with the new columns added to the original ones. The existing columns that are re-assigned will be overwritten.
    .set_index('date', append=True) # 'set_index' sets the index using one or more existing columns. By default yields a new object.
    .rename_axis(["game_id", "date"]) #Alter the name of the index or columns.
    .sort_index()) #Alter the name of the index or columns.

#The below code is used to understand the performance of games and analyse the results per game
delta = (by_game.home_rest - by_game.away_rest).dropna().astype(int)
ax = (delta.value_counts()
    .reindex(np.arange(delta.min(), delta.max() + 1), fill_value=0)
    .sort_index()
    .plot(kind='bar', color='k', width=.9, rot=0, figsize=(12, 6))
)
sns.despine() #Remove the top and right spines from plot(s).
ax.set(xlabel='Difference in Rest (Home - Away)', ylabel='Games'); # used to set the labels to the axes
